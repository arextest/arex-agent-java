CREATE TABLE IF NOT EXISTS RECORD_ServletEntrance(
id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
caseId VARCHAR(100) NOT NULL,
dataCreateTime TIMESTAMP,
appId VARCHAR(100),
path VARCHAR(100),
requestHeaders VARCHAR,
responseHeaders VARCHAR,
request VARCHAR,
response VARCHAR,
exceptionMessage VARCHAR,
mockerInfo VARCHAR,
createTime long
);
CREATE INDEX idx_caseId ON RECORD_ServletEntrance (caseId);
--
CREATE TABLE IF NOT EXISTS RECORD_Database(
id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
caseId VARCHAR(100) NOT NULL,
dataCreateTime TIMESTAMP,
appId VARCHAR(100),
dbName VARCHAR(100),
tables VARCHAR(100),
parameters VARCHAR(200),
sql VARCHAR,
response VARCHAR,
responseType VARCHAR(100),
exceptionMessage VARCHAR,
mockerInfo VARCHAR,
createTime long
);
CREATE INDEX idx_caseId ON RECORD_Database (caseId);
--
