CREATE TABLE IF NOT EXISTS RECORD_SERVLET_ENTRANCE(
id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
caseId VARCHAR(100) NOT NULL,
dataCreateTime TIMESTAMP,
appId VARCHAR(100),
path VARCHAR(100),
requestHeaders VARCHAR,
responseHeaders VARCHAR,
request VARCHAR,
response VARCHAR,
exceptionMessage VARCHAR,
mockerInfo VARCHAR,
createTime long
);
CREATE INDEX idx_caseId ON RECORD_SERVLET_ENTRANCE (caseId);
--
CREATE TABLE IF NOT EXISTS RECORD_DATABASE(
id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
caseId VARCHAR(100) NOT NULL,
dataCreateTime TIMESTAMP,
appId VARCHAR(100),
dbName VARCHAR(100),
tables VARCHAR(100),
parameters VARCHAR(200),
sql VARCHAR,
response VARCHAR,
responseType VARCHAR(100),
exceptionMessage VARCHAR,
mockerInfo VARCHAR,
createTime long
);
CREATE INDEX idx_caseId ON RECORD_DATABASE (caseId);
--
CREATE TABLE IF NOT EXISTS RECORD_DYNAMIC_CLASS(
id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
caseId VARCHAR(100) NOT NULL,
dataCreateTime TIMESTAMP,
clazzName VARCHAR(200),
operation VARCHAR(100),
operationKey VARCHAR,
operationResult VARCHAR,
resultClazz VARCHAR(200),
exceptionMessage VARCHAR,
mockerInfo VARCHAR,
createTime long
);
CREATE INDEX idx_caseId ON RECORD_DYNAMIC_CLASS (caseId);
--
CREATE TABLE IF NOT EXISTS REPLAY_SERVLET_ENTRANCE(
id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
replayId VARCHAR(100) NOT NULL,
caseId VARCHAR(100),
dataCreateTime TIMESTAMP,
appId VARCHAR(100),
path VARCHAR(100),
request VARCHAR,
response VARCHAR,
exceptionMessage VARCHAR,
mockerInfo VARCHAR,
createTime long
);
CREATE INDEX idx_replayId ON REPLAY_SERVLET_ENTRANCE (replayId);
--
CREATE TABLE IF NOT EXISTS REPLAY_DATABASE(
id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
replayId VARCHAR(100) NOT NULL,
caseId VARCHAR(100),
dataCreateTime TIMESTAMP,
appId VARCHAR(100),
dbName VARCHAR(100),
tables VARCHAR(100),
parameters VARCHAR(200),
sql VARCHAR,

createTime long
);
CREATE INDEX idx_replayId ON REPLAY_DATABASE (replayId);
--
CREATE TABLE IF NOT EXISTS REPLAY_DYNAMIC_CLASS(
id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
replayId VARCHAR(100) NOT NULL,
caseId VARCHAR(100),
dataCreateTime TIMESTAMP,

CREATE INDEX idx_replayId ON REPLAY_DYNAMIC_CLASS (replayId);
--
CREATE TABLE IF NOT EXISTS DIFF_RESULT(
id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
replayId VARCHAR(100) NOT NULL,
caseId VARCHAR(100),
category VARCHAR(50),
dataCreateTime TIMESTAMP,
recordDiff VARCHAR,
replayDiff VARCHAR
);
CREATE INDEX idx_replayId ON DIFF_RESULT (replayId);
